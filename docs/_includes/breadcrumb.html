{%- comment -%} Fil d'Ariane construit à partir de page.url. Le libellé de chaque niveau parent est récupéré depuis la page index correspondante (titre issu du H1 via jekyll-titles-from-headings), avec repli sur le slug humanisé si la page n'est pas trouvée. Masqué sur l'accueil. {%- endcomment -%} {%- unless page.url == '/' -%} {%- assign parts = page.url | split: '/' -%} {%- assign acc = '' -%}
  1. Accueil
    2. {%- for part in parts -%} {%- if part != '' -%} {%- assign acc = acc | append: '/' | append: part -%} {%- if forloop.last -%}
  2. {{ page.title }}
    3. {%- else -%} {%- assign dir_url = acc | append: '/' -%} {%- assign ancestor = site.pages | where: 'url', dir_url | first -%} {%- if ancestor -%} {%- assign label = ancestor.title -%} {%- else -%} {%- assign label = part | replace: '-', ' ' | capitalize -%} {%- endif -%}
  3. {{ label }}
    4. {%- endif -%} {%- endif -%} {%- endfor -%}
{%- endunless -%}
